<credential-repository xmlns="urn:X-quest.com:glue/credential-repository/3.0">

  <!-- just some default entries to get things rolling -->
  <resource-mappings>
    <resource-map alias="Toronto">
      <target-hosts>
        <target-host pattern="tor.*"/>
        <target-host pattern="rd-.*"/>
        <target-host pattern="lab-.*"/>
      </target-hosts>
    </resource-map>
    <resource-map alias="Zhuhai">
      <target-hosts>
        <target-host pattern="zhvm.*"/>
      </target-hosts>

    </resource-map>
  </resource-mappings>

  <!-- Policy examples -->
  <policies>
    <cache alias="Cache-For-An-Hour" critical="true" time-to-live-millis="3600000"/>
    <use-count alias="Use-3-Times" critical="true" use-count="3"/>
    <failure-rate alias="Three-Strikes" critical="true" max-fail-count="3" period-millis="300000"/>
  </policies>

  <!-- Credential Definitions -->
  <credentials>

    <login-password alias="UNIX" purpose="SSH" username="USERNAME" password="PASSWORD">
      <resource-aliases>
        <resource-alias name="Toronto"/>
      </resource-aliases>
      <policy-aliases>
        <policy-alias name="Three-Strikes"/>
        <policy-alias name="Use-3-Times"/>
      </policy-aliases>
    </login-password>

    <windows alias="Active-Directory" purpose="WMI" username="USERNAME" password="PASSWORD" domain="DOMAIN">
      <resource-aliases>
        <resource-alias name="Toronto"/>
        <resource-alias name="Zhuhai"/>
      </resource-aliases>
      <policy-aliases>
        <policy-alias name="Three-Strikes"/>
      </policy-aliases>
    </windows>

  </credentials>

</credential-repository>
